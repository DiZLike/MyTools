from django.views.generic import TemplateView
from django.conf import settings
from django.conf.urls.static import static
from django.http import HttpResponse

sitemaps = {
    'posts': PostSitemap,
    'categories': CategorySitemap,
    'static': StaticViewSitemap,
}

urlpatterns = [
    path('admin/', admin.site.urls),
    path('markdownx/', include('markdownx.urls')),
    path('', include('chat.urls')),
    path('', include('users.urls')),
    path('', include('blog.urls')),
    path('music/', include('music.urls')),
    path("robots.txt", TemplateView.as_view(template_name="robots.txt", content_type="text/plain")),
    path('sitemap.xml', sitemap, {'sitemaps': sitemaps}, name='django.contrib.sitemaps.views.sitemap'),

    # Подтверждение Яндекс.Вебмастера
    path("yandex_42fc5652e62276fa.html", lambda request: HttpResponse(
        '<html><head><meta http-equiv="Content-Type" content="text/html; charset=UTF-8"></head><body>Verification: 42fc5652e62276fa</body></html>',
        content_type="text/html; charset=utf-8"

<script>
document.addEventListener('DOMContentLoaded', function() {
    const likeBtn = document.querySelector('.like-btn');
    const dislikeBtn = document.querySelector('.dislike-btn');
    
    if (!likeBtn || !dislikeBtn) return;
    
    async function handleReaction(reactionType) {
        try {
            const response = await fetch('{% url "blog:post_react" post.slug %}', {
                method: 'POST',
                headers: {
                    'X-CSRFToken': '{{ csrf_token }}',
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: `reaction_type=${reactionType}`
            });
            
            const data = await response.json();
            
            if (data.success) {
                // Обновляем счетчики
                document.querySelector('.likes-count').textContent = data.likes_count;
                document.querySelector('.dislikes-count').textContent = data.dislikes_count;
                document.querySelector('.rating-value').textContent = data.rating;
                
                // Обновляем активные состояния кнопок
                likeBtn.classList.toggle('active', data.user_reaction === 'like');
                dislikeBtn.classList.toggle('active', data.user_reaction === 'dislike');
            }
        } catch (error) {
            console.error('Error:', error);
        }
    }
    
    likeBtn.addEventListener('click', () => handleReaction('like'));
    dislikeBtn.addEventListener('click', () => handleReaction('dislike'));
});
</script>
