// Z:\web\static\music\js\favorites.js

document.addEventListener('DOMContentLoaded', () => {
    // Инициализируем плеер
    if (!MusicPlayer.init()) {
        console.warn('Музыкальный плеер не инициализирован на странице избранного');
        return;
    }

    const state = MusicPlayer.getState();
    const elements = MusicPlayer.getElements();

    // Переопределяем обработчик окончания трека для автовоспроизведения следующего
    elements.onTrackEnded = () => {
        playNextTrack();
    };

    // Инициализация плейлиста
    const initPlaylist = () => {
        const playlist = [];
        document.querySelectorAll('.playlist-card[data-track-uid]').forEach(card => {
            playlist.push(card.dataset.trackUid);
        });
        MusicPlayer.updateState({ playlist });
    };
    
    initPlaylist();

    // === Специфичные функции для избранного ===
    
    // Получение данных трека из DOM
    const getTrackDataFromCard = (uid) => {
        const card = document.querySelector(`.playlist-card[data-track-uid="${uid}"]`);
        if (!card) return null;
        
        const img = card.querySelector('.cover-wrapper img');
        const artistEl = card.querySelector('.track-info h6');
        const titleEl = card.querySelector('.track-info p');
        
        if (!artistEl || !titleEl || !img) return null;
        
        return {
            artist: artistEl.textContent.trim(),
            title: titleEl.textContent.trim(),
            cover: img.src || '/static/img/default-cover.png'
        };
    };

    // Воспроизведение трека из избранного
    window.playFavoriteTrack = function(uid, artist, title, cover) {
        const currentState = MusicPlayer.getState();
        const index = currentState.playlist.indexOf(uid);
        if (index === -1) return;
        
        MusicPlayer.updateState({ 
            currentTrackUid: uid, 
            currentIndex: index 
        });
        
        // Обновляем UI
        if (elements.audioPlayerBar) elements.audioPlayerBar.style.display = 'block';
        if (elements.playerBarCover) elements.playerBarCover.src = cover || '/static/img/default-cover.png';
        if (elements.playerBarArtist) elements.playerBarArtist.textContent = artist || 'Неизвестный исполнитель';
        if (elements.playerBarTitle) elements.playerBarTitle.textContent = title || 'Без названия';
        if (elements.progressBar) elements.progressBar.value = 0;
        if (elements.currentTimeEl) elements.currentTimeEl.textContent = '0:00';
        if (elements.durationEl) elements.durationEl.textContent = '0:00';
        
        // Получаем URL для стриминга
        MusicPlayer.fetchCSRF('/music/api/stream/', { uid })
            .then(res => res.json())
            .then(data => {
                if (data.success && data.stream_url) {
                    MusicPlayer.updateState({ currentAudio: data.stream_url });
                    MusicPlayer.setAudioSource(data.stream_url);
                } else {
                    console.error('Не удалось получить URL для стриминга:', data.error);
                    alert(data.error || 'Не удалось получить URL для стриминга');
                }
            })
            .catch(err => {
                console.error('Ошибка получения URL:', err);
                alert('Ошибка при получении URL для стриминга');
            });
    };

    // Воспроизвести всё избранное
    window.playAllFavorites = function() {
        const currentState = MusicPlayer.getState();
        if (currentState.playlist.length > 0) {
            const firstTrack = getTrackDataFromCard(currentState.playlist[0]);
            if (firstTrack) {
                window.playFavoriteTrack(
                    currentState.playlist[0], 
                    firstTrack.artist, 
                    firstTrack.title, 
                    firstTrack.cover
                );
            }
        }
    };

    // Следующий трек
    function playNextTrack() {
        const currentState = MusicPlayer.getState();
        if (currentState.currentIndex < currentState.playlist.length - 1) {
            const nextUid = currentState.playlist[currentState.currentIndex + 1];
            const trackData = getTrackDataFromCard(nextUid);
            if (trackData) {
                window.playFavoriteTrack(nextUid, trackData.artist, trackData.title, trackData.cover);
            }
        } else {
            MusicPlayer.closePlayer();
        }
    }

    // Предыдущий трек
    function playPreviousTrack() {
        const currentState = MusicPlayer.getState();
        if (currentState.currentIndex > 0) {
            const prevUid = currentState.playlist[currentState.currentIndex - 1];
            const trackData = getTrackDataFromCard(prevUid);
            if (trackData) {
                window.playFavoriteTrack(prevUid, trackData.artist, trackData.title, trackData.cover);
            }
        }
    }

    // Удаление из избранного
    window.removeFromFavorites = function(uid, cardElement) {
        if (!confirm('Удалить трек из избранного?')) return;
        
        MusicPlayer.fetchCSRF(`/music/api/favorite/${uid}/remove/`, {})
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    const currentState = MusicPlayer.getState();
                    
                    // Если удаляем текущий трек - закрываем плеер
                    if (currentState.currentTrackUid === uid) {
                        MusicPlayer.closePlayer();
                    }
                    
                    // Обновляем плейлист
                    const index = currentState.playlist.indexOf(uid);
                    if (index > -1) {
                        const newPlaylist = [...currentState.playlist];
                        newPlaylist.splice(index, 1);
                        
                        let newIndex = currentState.currentIndex;
                        if (currentState.currentIndex > index) {
                            newIndex--;
                        } else if (currentState.currentIndex === index) {
                            newIndex = -1;
                        }
                        
                        MusicPlayer.updateState({ 
                            playlist: newPlaylist,
                            currentIndex: newIndex,
                            currentTrackUid: newIndex === -1 ? null : currentState.currentTrackUid
                        });
                    }
                    
                    // Анимация удаления
                    if (cardElement) {
                        cardElement.style.transition = 'all 0.3s ease';
                        cardElement.style.opacity = '0';
                        cardElement.style.transform = 'translateX(50px)';
                        
                        setTimeout(() => {
                            cardElement.remove();
                            
                            const newState = MusicPlayer.getState();
                            const totalText = document.querySelector('.favorites-header p');
                            if (totalText) {
                                totalText.textContent = `${newState.playlist.length} ${getTrackWord(newState.playlist.length)} в вашей коллекции`;
                            }
                            
                            updateTrackNumbers();
                            
                            if (newState.playlist.length === 0) {
                                location.reload();
                            }
                        }, 300);
                    }
                } else if (data.error) {
                    alert(data.error);
                }
            })
            .catch(err => {
                console.error('Ошибка при удалении из избранного:', err);
                alert('Произошла ошибка при удалении трека');
            });
    };

    // Вспомогательные функции
    function updateTrackNumbers() {
        document.querySelectorAll('.playlist-card .track-number').forEach((numEl, index) => {
            numEl.textContent = index + 1;
        });
    }

    function getTrackWord(count) {
        const lastDigit = count % 10;
        const lastTwoDigits = count % 100;
        
        if (lastDigit === 1 && lastTwoDigits !== 11) return 'трек';
        if ([2, 3, 4].includes(lastDigit) && ![12, 13, 14].includes(lastTwoDigits)) return 'трека';
        return 'треков';
    }

    // === Глобальные функции для совместимости ===
    window.togglePlayPause = () => MusicPlayer.togglePlayPause();
    window.seekAudio = (seconds) => MusicPlayer.seekAudio(seconds);
    window.closePlayer = () => MusicPlayer.closePlayer();
    window.playTrack = (...args) => MusicPlayer.playTrack(...args);
    window.downloadTrack = (...args) => MusicPlayer.downloadTrack(...args);
    window.toggleFavorite = (...args) => MusicPlayer.toggleFavorite(...args);
    window.quickSearch = (query) => MusicPlayer.quickSearch(query);
    window.playNext = playNextTrack;
    window.playPrevious = playPreviousTrack;
    
    // Для обратной совместимости с playLocalTrack
    window.playLocalTrack = function(uid, artist, title, cover) {
        MusicPlayer.fetchCSRF('/music/api/stream/', { uid })
            .then(res => res.json())
            .then(data => {
                if (data.success && data.stream_url) {
                    MusicPlayer.playTrack(data.stream_url, artist, title, cover, null, uid);
                }
            })
            .catch(err => console.error('Ошибка:', err));
    };
});

// Z:\web\static\music\js\player-core.js

/**
 * Music Player Core - общий модуль для управления аудиоплеером
 * Версия: 1.0.0
 */
const MusicPlayer = (() => {
    // Приватные переменные
    let elements = {};
    let state = {
        currentAudio: null,
        isSeeking: false,
        seekRecovery: null,
        currentTrackUid: null,
        playlist: [],
        currentIndex: -1,
        currentButton: null
    };

    // === Утилиты ===
    const formatTime = (seconds) => {
        if (!isFinite(seconds) || seconds < 0) return '0:00';
        const mins = Math.floor(seconds / 60);
        const secs = Math.floor(seconds % 60);
        return `${mins}:${secs.toString().padStart(2, '0')}`;
    };

    const getCookie = (name) => {
        const match = document.cookie.match(new RegExp(`(^| )${name}=([^;]+)`));
        return match ? decodeURIComponent(match[2]) : null;
    };

    const escapeHtml = (text) => {
        if (!text) return '';
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    };

    const fetchCSRF = (url, body, options = {}) => {
        const formData = new FormData();
        Object.entries(body).forEach(([key, value]) => formData.append(key, value));
        
        return fetch(url, {
            method: 'POST',
            body: formData,
            headers: { 'X-CSRFToken': getCookie('csrftoken'), ...options.headers }
        });
    };

    // === Инициализация ===
    const init = (customElements = {}) => {
        // Кэшируем базовые элементы плеера
        const defaultElements = {
            audioPlayerBar: document.getElementById('audioPlayerBar'),
            audio: document.getElementById('audioPlayer'),
            playPauseBtn: document.getElementById('playPauseBtn'),
            progressBar: document.getElementById('progressBar'),
            currentTimeEl: document.getElementById('currentTime'),
            durationEl: document.getElementById('duration'),
            playerBarCover: document.getElementById('playerBarCover'),
            playerBarArtist: document.getElementById('playerBarArtist'),
            playerBarTitle: document.getElementById('playerBarTitle')
        };

        elements = { ...defaultElements, ...customElements };

        // Проверяем наличие аудио элемента
        if (!elements.audio) {
            console.warn('MusicPlayer: аудио элемент не найден');
            return false;
        }

        // Привязываем события плеера
        bindEvents();
        return true;
    };

    const bindEvents = () => {
        if (!elements.audio) return;

        elements.audio.addEventListener('timeupdate', updateProgress);
        elements.audio.addEventListener('loadedmetadata', updateDuration);
        elements.audio.addEventListener('ended', onTrackEnded);
        elements.audio.addEventListener('play', () => setPlayIcon(false));
        elements.audio.addEventListener('pause', () => setPlayIcon(true));
        
        elements.audio.addEventListener('seeking', () => {
            state.isSeeking = true;
        });
        
        elements.audio.addEventListener('seeked', () => {
            state.isSeeking = false;
            if (state.seekRecovery !== null) {
                elements.audio.currentTime = state.seekRecovery;
                state.seekRecovery = null;
            }
            updateProgress();
        });

        // Обработка ошибок
        elements.audio.addEventListener('error', (e) => {
            console.error('MusicPlayer: ошибка аудио:', e);
            if (elements.audio.error) {
                console.error('Код ошибки:', elements.audio.error.code);
                console.error('Сообщение:', elements.audio.error.message);
            }
        });

        elements.audio.addEventListener('stalled', () => {
            console.warn('MusicPlayer: воспроизведение приостановлено, буферизация...');
        });

        elements.audio.addEventListener('waiting', () => {
            console.log('MusicPlayer: ожидание данных...');
        });

        // Обработчик прогресс-бара
        if (elements.progressBar) {
            let seekTimeout;
            
            elements.progressBar.addEventListener('input', (e) => {
                if (elements.audio?.duration && isFinite(elements.audio.duration)) {
                    state.isSeeking = true;
                    const seekTime = (e.target.value / 100) * elements.audio.duration;
                    state.seekRecovery = seekTime;
                    
                    if (seekTimeout) clearTimeout(seekTimeout);
                    
                    seekTimeout = setTimeout(() => {
                        if (elements.audio) {
                            elements.audio.currentTime = seekTime;
                            if (elements.currentTimeEl) {
                                elements.currentTimeEl.textContent = formatTime(seekTime);
                            }
                        }
                    }, 50);
                }
            });
            
            elements.progressBar.addEventListener('change', (e) => {
                if (elements.audio?.duration && isFinite(elements.audio.duration)) {
                    const seekTime = (e.target.value / 100) * elements.audio.duration;
                    state.seekRecovery = seekTime;
                    elements.audio.currentTime = seekTime;
                    
                    setTimeout(() => {
                        state.isSeeking = false;
                        state.seekRecovery = null;
                    }, 200);
                }
            });
        }
    };

    const updateProgress = () => {
        if (!state.isSeeking && elements.audio?.duration && isFinite(elements.audio.duration)) {
            const value = (elements.audio.currentTime / elements.audio.duration) * 100;
            if (elements.progressBar) {
                elements.progressBar.value = isFinite(value) ? value : 0;
            }
            if (elements.currentTimeEl) {
                elements.currentTimeEl.textContent = formatTime(elements.audio.currentTime);
            }
        }
    };

    const updateDuration = () => {
        if (elements.audio?.duration && isFinite(elements.audio.duration)) {
            if (elements.durationEl) {
                elements.durationEl.textContent = formatTime(elements.audio.duration);
            }
            if (elements.progressBar) {
                elements.progressBar.value = 0;
            }
            if (elements.currentTimeEl) {
                elements.currentTimeEl.textContent = '0:00';
            }
        } else {
            setTimeout(updateDuration, 500);
        }
    };

    const onTrackEnded = () => {
        // По умолчанию сбрасываем иконки
        setPlayIcon(true);
        if (elements.progressBar) elements.progressBar.value = 0;
        if (elements.currentTimeEl) elements.currentTimeEl.textContent = '0:00';
        state.currentAudio = null;
        
        // Вызываем пользовательский обработчик если есть
        if (typeof elements.onTrackEnded === 'function') {
            elements.onTrackEnded();
        }
    };

    const setPlayIcon = (isPaused) => {
        const icon = elements.playPauseBtn?.querySelector('i');
        if (icon) {
            icon.className = `bi bi-${isPaused ? 'play' : 'pause'}-fill`;
        }
        
        // Обновляем кнопку текущего трека если есть
        if (state.currentButton) {
            state.currentButton.innerHTML = `<i class="bi bi-${isPaused ? 'play' : 'pause'}-fill"></i> ${isPaused ? 'Слушать' : 'Пауза'}`;
        }
        
        // Обновляем иконки в плейлисте
        updatePlaylistIcons();
    };

    const updatePlaylistIcons = () => {
        document.querySelectorAll('.play-overlay i').forEach(icon => {
            icon.className = 'bi bi-play-fill';
        });
        
        if (state.currentTrackUid) {
            const currentCard = document.querySelector(`.playlist-card[data-track-uid="${state.currentTrackUid}"]`);
            if (currentCard) {
                const icon = currentCard.querySelector('.play-overlay i');
                if (icon) {
                    icon.className = elements.audio?.paused ? 'bi bi-play-fill' : 'bi bi-pause-fill';
                }
            }
        }
    };

    const setAudioSource = (src) => {
        if (!elements.audio) return;
        
        elements.audio.removeAttribute('src');
        elements.audio.load();
        elements.audio.src = src;
        elements.audio.load();
        
        const playPromise = elements.audio.play();
        if (playPromise) {
            playPromise.catch(err => {
                console.error('MusicPlayer: ошибка воспроизведения:', err);
                if (state.currentButton) {
                    state.currentButton.innerHTML = '<i class="bi bi-play-fill"></i> Слушать';
                }
            });
        }
    };

    // === Публичные методы ===
    return {
        // Геттеры
        getState: () => state,
        getElements: () => elements,
        
        // Инициализация
        init,
        
        // Утилиты
        formatTime,
        getCookie,
        escapeHtml,
        fetchCSRF,
        
        // Управление плеером
        togglePlayPause: () => {
            if (!elements.audio) return;
            
            if (elements.audio.paused) {
                elements.audio.play().catch(err => {
                    console.error('MusicPlayer: ошибка воспроизведения:', err);
                });
            } else {
                elements.audio.pause();
            }
        },
        
        seekAudio: (seconds) => {
            if (!elements.audio?.duration || !isFinite(elements.audio.duration)) return;
            const newTime = Math.min(Math.max(elements.audio.currentTime + seconds, 0), elements.audio.duration);
            elements.audio.currentTime = newTime;
        },
        
        closePlayer: () => {
            if (!elements.audio) return;
            
            elements.audio.pause();
            elements.audio.removeAttribute('src');
            elements.audio.load();
            elements.audio.currentTime = 0;
            if (elements.audioPlayerBar) {
                elements.audioPlayerBar.style.display = 'none';
            }
            state.currentTrackUid = null;
            state.currentIndex = -1;
            state.currentAudio = null;
            state.currentButton = null;
            state.seekRecovery = null;
            state.playlist = [];

            document.querySelectorAll('.play-overlay i, .btn-outline-light i.bi-pause-fill').forEach(icon => {
                icon.className = 'bi bi-play-fill';
                const btn = icon.parentElement;
                if (btn?.tagName === 'BUTTON') {
                    btn.innerHTML = '<i class="bi bi-play-fill"></i> Слушать';
                }
            });
        },
        
        setAudioSource,
        
        // Воспроизведение трека
        playTrack: function(url, artist, title, cover, button, uid = null) {
            if (!elements.audio) return;
            
            // Если это тот же трек - переключаем play/pause
            if (state.currentAudio === url && elements.audio.src) {
                this.togglePlayPause();
                return;
            }
            
            // Обновляем UI
            if (elements.audioPlayerBar) elements.audioPlayerBar.style.display = 'block';
            if (elements.playerBarCover) elements.playerBarCover.src = cover || '/static/img/default-cover.png';
            if (elements.playerBarArtist) elements.playerBarArtist.textContent = artist || 'Неизвестный исполнитель';
            if (elements.playerBarTitle) elements.playerBarTitle.textContent = title || 'Без названия';
            if (elements.progressBar) elements.progressBar.value = 0;
            if (elements.currentTimeEl) elements.currentTimeEl.textContent = '0:00';
            if (elements.durationEl) elements.durationEl.textContent = '0:00';
            
            // Сбрасываем иконки всех кнопок
            document.querySelectorAll('.btn-outline-light i.bi-pause-fill').forEach(icon => {
                icon.className = 'bi bi-play-fill';
                const btn = icon.parentElement;
                if (btn?.tagName === 'BUTTON') {
                    btn.innerHTML = '<i class="bi bi-play-fill"></i> Слушать';
                }
            });
            
            state.currentButton = button;
            state.currentAudio = url;
            
            if (button) {
                button.innerHTML = '<i class="bi bi-hourglass-split"></i> Загрузка...';
            }
            
            // Определяем источник
            if (uid) {
                fetchCSRF('/music/api/stream/', { uid, url })
                    .then(res => res.json())
                    .then(data => {
                        setAudioSource(data.success && data.stream_url ? data.stream_url : url);
                    })
                    .catch(() => setAudioSource(url));
            } else {
                fetchCSRF('/music/api/stream/', { url })
                    .then(res => res.json())
                    .then(data => {
                        setAudioSource(data.success && data.stream_url ? data.stream_url : url);
                    })
                    .catch(() => setAudioSource(url));
            }
        },

        downloadTrack: function(url, artist, title, uid, coverUrl, button) {
            if (!button) return;
            
            const originalHTML = button.innerHTML;
            button.disabled = true;
            button.innerHTML = '<i class="bi bi-hourglass-split"></i>';
            
            fetchCSRF('/music/api/download/', { url, artist, title, uid, cover_url: coverUrl })
                .then(response => {
                    if (response.ok && response.headers.get('content-type')?.includes('audio')) {
                        return response.blob().then(blob => {
                            const blobUrl = window.URL.createObjectURL(blob);
                            const a = document.createElement('a');
                            a.href = blobUrl;
                            a.download = `${artist} - ${title}_dlike.ru.mp3`;
                            document.body.appendChild(a);
                            a.click();
                            document.body.removeChild(a);
                            window.URL.revokeObjectURL(blobUrl);
                            
                            button.innerHTML = '<i class="bi bi-check-lg"></i> Готово';
                            setTimeout(() => {
                                button.disabled = false;
                                button.innerHTML = originalHTML;
                            }, 2000);
                        });
                    }
                    return response.json();
                })
                .then(data => {
                    if (data && data.error) {
                        button.disabled = false;
                        button.innerHTML = originalHTML;
                        alert(data.error);
                    }
                })
                .catch(() => {
                    button.disabled = false;
                    button.innerHTML = originalHTML;
                    alert('Ошибка при скачивании');
                });
        },

                download_url: downloadUrl

                        window.location.href = '/login/';
                        return;

                        button.className = `btn btn-sm ${isFav ? 'btn-danger' : 'btn-outline-danger'} favorite-btn`;
                        button.innerHTML = `<i class="bi bi-${isFav ? 'heart-fill' : 'heart'}"></i>`;
                        
                        if (isFav && data.track?.has_file) {
                            console.log('Трек добавлен в избранное и доступен офлайн');
                        } else if (isFav) {
                            console.log('Трек добавлен в избранное (файл загружается)');
                        }
                    }
                })
                .catch(err => {
                    console.error('MusicPlayer: ошибка при добавлении в избранное:', err);
                });
        },
        
        // Быстрый поиск
        quickSearch: function(query, searchPageUrl = '/music/') {
            const searchInput = document.getElementById('searchInput');
            if (searchInput) {
                searchInput.value = query;
                if (typeof window.performSearch === 'function') {
                    window.performSearch();
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                }
            } else {
                window.location.href = `${searchPageUrl}?q=${encodeURIComponent(query)}`;
            }
        },
        
        // Обновление состояния
        updateState: (newState) => {
            state = { ...state, ...newState };
        },
        
        // Сброс иконок кнопок
        resetAllButtons: () => {
            document.querySelectorAll('.btn-outline-light i.bi-pause-fill').forEach(icon => {
                icon.className = 'bi bi-play-fill';
                const btn = icon.parentElement;
                if (btn?.tagName === 'BUTTON') {
                    btn.innerHTML = '<i class="bi bi-play-fill"></i> Слушать';
                }
            });
        }
    };
})();

// Z:\web\static\music\js\player.js

document.addEventListener('DOMContentLoaded', () => {
    // Инициализируем плеер
    if (!MusicPlayer.init()) {
        console.warn('Музыкальный плеер не инициализирован - элементы не найдены');
        return;
    }

    // Кэшируем дополнительные элементы поиска
    const searchElements = {
        searchInput: document.getElementById('searchInput'),
        searchBtn: document.getElementById('searchBtn'),
        loadingIndicator: document.getElementById('loadingIndicator'),
        searchResults: document.getElementById('searchResults'),
        errorMessage: document.getElementById('errorMessage'),
        initialState: document.getElementById('initialState'),
        tracksList: document.getElementById('tracksList'),
        resultsCount: document.getElementById('resultsCount'),
        searchQuery: document.getElementById('searchQuery')
    };

    // === Поиск ===
    const hideAll = () => {
        ['loadingIndicator', 'searchResults', 'errorMessage', 'initialState'].forEach(id => {
            const el = searchElements[id];
            if (el) el.style.display = 'none';
        });
    };

    const showError = (message) => {
        hideAll();
        if (searchElements.errorMessage) {
            searchElements.errorMessage.textContent = message;
            searchElements.errorMessage.style.display = 'block';
        }
        setTimeout(() => {
            if (searchElements.errorMessage) {
                searchElements.errorMessage.style.display = 'none';
            }
            if (searchElements.initialState) {
                searchElements.initialState.style.display = 'block';
            }
        }, 4000);
    };

    const displayResults = (tracks, query) => {
        if (!searchElements.tracksList) return;
        
        if (searchElements.resultsCount) searchElements.resultsCount.textContent = tracks.length;
        if (searchElements.searchQuery) searchElements.searchQuery.textContent = query;
        searchElements.tracksList.innerHTML = '';

        tracks.forEach((track, index) => {
            const col = document.createElement('div');
            col.className = 'col-md-6 col-lg-4';
            
            const coverUrl = track.cover_url || '/static/img/default-cover.png';
            const safe = {
                downloadUrl: MusicPlayer.escapeHtml(track.download_url || ''),
                artist: MusicPlayer.escapeHtml(track.artist || 'Неизвестный исполнитель'),
                title: MusicPlayer.escapeHtml(track.title || 'Без названия'),
                coverUrl: MusicPlayer.escapeHtml(coverUrl),
                uid: MusicPlayer.escapeHtml(track.uid || '')
            };
            
            col.innerHTML = `
                <div class="card mb-3 fade-in" style="animation-delay: ${index * 0.05}s">
                    <div class="card-body">
                        <div class="d-flex align-items-start gap-3">
                            <img src="${coverUrl}" class="rounded track-cover" 
                                 onerror="this.src='/static/img/default-cover.png'" 
                                 alt="${safe.artist} - ${safe.title}">
                            <div class="flex-grow-1 min-width-0">
                                <h6 class="card-title mb-1 text-truncate">${safe.artist}</h6>
                                <p class="card-text text-secondary mb-2 text-truncate">${safe.title}</p>
                                <span class="badge bg-secondary">${track.duration || 'Неизвестно'}</span>
                            </div>
                        </div>
                        <div class="mt-3 d-flex gap-2">
                            <button class="btn btn-sm btn-outline-light flex-grow-1" 
                                    onclick="window.playTrack('${safe.downloadUrl}', '${safe.artist}', '${safe.title}', '${safe.coverUrl}', this, '${safe.uid}')">
                                <i class="bi bi-play-fill"></i> Слушать
                            </button>
                            <button class="btn btn-sm btn-primary flex-grow-1" 
                                    onclick="window.downloadTrack('${safe.downloadUrl}', '${safe.artist}', '${safe.title}', '${safe.uid}', '${safe.coverUrl}', this)">
                                <i class="bi bi-download"></i> Скачать
                            </button>
                            <button class="btn btn-sm ${track.is_favorite ? 'btn-danger' : 'btn-outline-danger'} favorite-btn" 
                                    onclick="window.toggleFavorite('${safe.uid}', '${safe.artist}', '${safe.title}', '${safe.coverUrl}', this, '${safe.downloadUrl}')">
                                <i class="bi ${track.is_favorite ? 'bi-heart-fill' : 'bi-heart'}"></i>
                            </button>
                        </div>
                    </div>
                </div>`;
            
            searchElements.tracksList.appendChild(col);
        });

        if (searchElements.searchResults) searchElements.searchResults.style.display = 'block';
        if (searchElements.initialState) searchElements.initialState.style.display = 'none';
    };

    // Выполнение поиска
    window.performSearch = () => {
        if (!searchElements.searchInput) return;
        
        const query = searchElements.searchInput.value.trim();
        
        if (query.length < 2) {
            showError('Введите минимум 2 символа');
            return;
        }

        hideAll();
        if (searchElements.loadingIndicator) searchElements.loadingIndicator.style.display = 'block';

        fetch(`/music/api/search/?q=${encodeURIComponent(query)}`)
            .then(res => res.json())
            .then(data => {
                if (searchElements.loadingIndicator) searchElements.loadingIndicator.style.display = 'none';
                data.success && data.tracks.length > 0 
                    ? displayResults(data.tracks, query)
                    : showError(data.error || 'Ничего не найдено');
            })
            .catch(() => {
                if (searchElements.loadingIndicator) searchElements.loadingIndicator.style.display = 'none';
                showError('Ошибка соединения');
            });
    };

    // Привязка событий поиска
    if (searchElements.searchBtn) {
        searchElements.searchBtn.addEventListener('click', window.performSearch);
    }
    
    if (searchElements.searchInput) {
        searchElements.searchInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') window.performSearch();
        });
    }

    // === Глобальные функции для совместимости ===
    window.togglePlayPause = () => MusicPlayer.togglePlayPause();
    window.seekAudio = (seconds) => MusicPlayer.seekAudio(seconds);
    window.closePlayer = () => MusicPlayer.closePlayer();
    window.playTrack = (...args) => MusicPlayer.playTrack(...args);
    window.downloadTrack = (...args) => MusicPlayer.downloadTrack(...args);
    window.toggleFavorite = (...args) => MusicPlayer.toggleFavorite(...args);
    window.quickSearch = (query) => MusicPlayer.quickSearch(query);
    
    // Специфичная функция для страницы поиска
    window.playLocalTrack = function(uid, artist, title, cover) {
        MusicPlayer.fetchCSRF('/music/api/stream/', { uid })
            .then(res => res.json())
            .then(data => {
                if (data.success && data.stream_url) {
                    MusicPlayer.playTrack(data.stream_url, artist, title, cover, null, uid);
                } else {
                    console.error('Не удалось получить URL для стриминга');
                }
            })
            .catch(err => console.error('Ошибка получения URL:', err));
    };
});
